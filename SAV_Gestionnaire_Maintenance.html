🔄 migrate_appuse_module.py — Migrer app.use() → instance.use() par module
# migrate_appuse_module.py — Migration ciblée app.use() → instance.use()
# Modifier MODULE_KEYWORD pour cibler un module spécifique
import re

TS_FILE = r"D:\OneDrive_Perso\OneDrive\Documents\Micro_Logiciel\Documentation\Frontend\micro_logiciel_frontend_nextjs\api\src\custom-routes.ts"
MODULE_KEYWORD = "factures"  # ← changer ici : "devis", "reglements", "clients"...

with open(TS_FILE, 'r', encoding='utf-8') as f:
    content = f.read()

# Pattern : app.use('/MODULE_KEYWORD...' ou app.use(`/MODULE_KEYWORD...
pattern = rf"(app\.use\(['\"`]/{re.escape(MODULE_KEYWORD)})"
replacement = rf"instance.use('{MODULE_KEYWORD})"

count = len(re.findall(pattern, content))
print(f"Occurrences trouvées pour '{MODULE_KEYWORD}': {count}")

if count == 0:
    print("Rien à migrer. Vérifiez MODULE_KEYWORD.")
    exit(0)

# Remplacement safe ligne par ligne
lines = content.split('\n')
new_lines = []
migrated = 0
for line in lines:
    if re.search(pattern, line):
        new_line = re.sub(r'\bapp\.use\(', 'instance.use(', line)
        new_lines.append(new_line)
        migrated += 1
        print(f"  ✅ Migré: {line.strip()[:70]}")
    else:
        new_lines.append(line)

with open(TS_FILE, 'w', encoding='utf-8') as f:
    f.write('\n'.join(new_lines))

print(f"\n✅ {migrated} occurrences migrées pour module '{MODULE_KEYWORD}'")
print("Relancer: docker compose build api --no-cache && docker compose up -d api && docker compose restart nginx")

# scan_routes.py — Scanner les routes API Anim'Gest
import urllib.request, json, time

BASE = "http://localhost"
ROUTES = [
    "/api/bilan", "/api/clients", "/api/animaux", "/api/seances",
    "/api/devis", "/api/factures", "/api/reglements", "/api/avoirs",
    "/api/achats", "/api/commandes-fourn", "/api/planning",
    "/api/agenda", "/api/races", "/api/tiers", "/api/inventaires",
    "/api/tarification", "/api/registre-sanitaire", "/api/statistiques",
    "/api/parametres/modules", "/api/dashboard/stats",
]

ok, errors = [], []
for route in ROUTES:
    try:
        req = urllib.request.urlopen(BASE + route, timeout=5)
        status = req.status
    except urllib.error.HTTPError as e:
        status = e.code
    except Exception as e:
        status = f"ERR: {e}"

    icon = "✅" if status == 200 else ("⚠️" if status == 401 else "❌")
    print(f"{icon} {status:3} {route}")
    (ok if status in [200, 401] else errors).append(route)

print(f"\n✅ OK: {len(ok)}/{len(ROUTES)} | ❌ Erreurs: {len(errors)}")
if errors: print(f"Routes en erreur : {errors}")
